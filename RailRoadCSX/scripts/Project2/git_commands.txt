git reset --hard <commidId> && git clean -f

http://stackoverflow.com/questions/6591213/how-do-i-rename-a-local-git-branch

# Create a new branch from an old commit and checkout that branch
git checkout -b <new_branch> <old_commit_id>
example:
git checkout -b latest_master abcdef123

# Remove old master branch locally and also from bitbucket
git branch -D master

# Rename latest_master to master 
git branch -m latest_master master

# Finally, push to origin (bitbucket)
git push origin master
